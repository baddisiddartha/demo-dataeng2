import requests
import pandas as pd

url = "https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd"

response = requests.get(url)
data = response.json()

price = data["bitcoin"]["usd"]

df = pd.DataFrame({
    "coin": ["bitcoin"],
    "price_usd": [price]
})

df.to_csv("bitcoin.csv", index=False)

print("CSV created successfully")
